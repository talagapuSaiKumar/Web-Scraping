def find_min_max_in(col):
    
    top = imdb_movies[col].idxmax()
    top_df = pd.DataFrame(imdb_movies.loc[top])
    
    bottom = imdb_movies[col].idxmin()
    bottom_df = pd.DataFrame(imdb_movies.loc[bottom])
    
    info_df = pd.concat([top_df, bottom_df], axis=1)
    return info_df
